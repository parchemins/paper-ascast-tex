$O_p$ \tcp*[r]{set of neighbors and weights}

$d \leftarrow \infty$ \tcp*[r]{lowest distance to $q$}


\BlankLine

\Function{\textup{Add ( )}} {
  \textup{receiveAdd($p$, $0$)} \label{line:lowestbound}
}

\BlankLine

\Function{\textup{receiveAdd($s'$, $d'$)}} {
  \If { $d' < d$ } {
    $d \leftarrow d'$ \;
    $s \leftarrow s'$ \;

    \lForEach {$\langle n, w_{pn} \rangle \in \mathcal{O}_p$} {
      \textup{send$_n$($s', d' + w_{pn}$)} \label{line:accumulator}
    }
    
  }

}
