\Function{\textup{receiveAdd($s'$, $d'$)} \tcp*[f]{$r_p(\alpha_{s'}^{d'})$}} {
  \If { $d' < d$ } {
      $d \leftarrow d'$ \;
      $s \leftarrow s'$ \tcp*[r]{$d_p(\alpha_{s'}^{d'})$}

      \ForEach {$\langle n, w_{pn} \rangle \in \mathcal{O}_p$} {
          \textup{send$_n$($s', d' + w_{pn}$)} \label{line:accumulator}
          \tcp*[r]{$b_p(\alpha_{s'}^{d'})$}
       }      

  }
}
