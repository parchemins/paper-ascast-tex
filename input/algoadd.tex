
\SetKwProg{Function}{func}{}{}

\small

\DontPrintSemicolon
\LinesNumbered

$\mathcal{O}_p$ \tcp*[r]{set of neighbors and weights}
$s$ \tcp*[r]{best source of partition}
$d \leftarrow \infty$ \tcp*[r]{lowest distance to $q$}


\BlankLine

\Function{\textup{Add ( )} \tcp*[f]{$\alpha_p^0$}} {
  \textup{receiveAdd($p$, $0$)} \label{line:lowestbound}
}

\BlankLine

\Function{\textup{receiveAdd($s'$, $d'$)} \tcp*[f]{$r(\alpha_{s'}^{d'})$}} {
  \lIf { $d' < d$ } {\textup{deliverAdd($s'$, $d'$)}}
}

\BlankLine

\Function{\textup{deliverAdd($s'$, $d'$)} \tcp*[f]{$d(\alpha_{s'}^{d'})$}} {
  $d \leftarrow d'$ \;
  $s \leftarrow s'$ \;

  \lForEach {$\langle n, w_{pn} \rangle \in \mathcal{O}_p$} {
    \textup{send$_n$($s', d' + w_{pn}$)} \label{line:accumulator}
  }
    
}
