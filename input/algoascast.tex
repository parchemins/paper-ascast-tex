       \lIf {$\pi' = \varnothing$} {send$_n$(\smash{$\delta_{s', c'}$)}

     }

  }\uElseIf{$A_{s, c}^{d, \pi} \neq \alpha_{\infty, 0}^{\infty, \varnothing}$} {
     \textup{send$_q$($\alpha_{s, c}^{d + W_{pq}, \pi \cup p}$)}

  }
